// fetch syntax
const user = { email, password };
fetch("http:localhost:8080/Player/login", {
	method: 'POST',
	headers: { 'Content-Type': 'application/json' },
	body: JSON.stringify(user)
})
.then(res => res.json())
.then((result) => {
	if (result.payload != 0) {
		fetch(`http:localhost:8080/Player/${result.payload}`)
		.then(res => res.json())
		.then((resultData) => {
			console.log(resultData);
		})
	}
	console.log(result);
})
